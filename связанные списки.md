```cpp
struct ListNode {
    int val;
    ListNode* next;
    ListNode() : val(0), next(nullptr) {}
    ListNode(int x) : val(x), next(nullptr) {}
    ListNode(int x, ListNode* next) : val(x), next(next) {}
    
};

```

```cpp
static ListNode* SetBigDig(int bg_d) {
    if (bg_d == 0) return new ListNode(0);

    ListNode dummy(0);        // фиктивная голова на стеке
    ListNode* tail = &dummy;  // хвост списка

    while (bg_d != 0) {
        int digit = bg_d % 10;
        bg_d /= 10;

        tail->next = new ListNode(digit);
        tail = tail->next;
    }
    return dummy.next; <-- вот здесь может быть утечка, если проебать эту ссылку 
}
```
